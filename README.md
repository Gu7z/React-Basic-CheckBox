## Install

```bash
npm install --save react-simple-checkbox
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-simple-checkbox'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
